 73     /**
 74      * The method that fills the randoms array in the program. It calls the
 75      * random(int low, int high) method to generate the random integers.
 76      *
 77      * @param low - The minimum value for the random integers.
 78      * @param high - The maximum value for the random integers.
 79      * @param size - The size of the array storing the random integers.
 80      * @return - Returns a randomly generated integer until the array is filled.
 81      */
 82     private static int[] random(int low, int high, int size) {
 83 
 84         int[] numbers = new int[size];
 85 
 86         for (int i = 0; i < size; i++) {
 87             numbers[i] = random(low, high);
 88         }
 89         return numbers;
 90     }

 92     /**
 93      * The method that generates the random integers.
 94      *
 95      * @param low - The minimum value for the integers.
 96      * @param high - The maximum value for the integers.
 97      * @return - Returns a randomly generated integer.
 98      */
 99     private static int random(int low, int high) {
100         double seed = Math.random();
101         double L = (double) low;
102         double H = (double) high;
103         double number = (H - L + 1) * seed + L;
104         return (int) number;
105     }
